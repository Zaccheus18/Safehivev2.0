# Function to load and return neural network model 
def load_model():
    json_file = open('models/Model.json', 'r')
    loaded_model_json = json_file.read()
    json_file.close()
    loaded_model = model_from_json(loaded_model_json)
    loaded_model.load_weights("weights/model_A_weights.h5")
    return loaded_model

# Function to load,normalize and return image 
def create_img(path):
    print(path)
    
    im = Image.open(path).convert('RGB')
    
    im = np.array(im)
    
    im = im/255.0
    
    im[:,:,0]=(im[:,:,0]-0.485)/0.229
    im[:,:,1]=(im[:,:,1]-0.456)/0.224
    im[:,:,2]=(im[:,:,2]-0.406)/0.225


    im = np.expand_dims(im,axis  = 0)
    return im

# Function to load image,predict heat map, generate count and return (count , image , heat map)
def predict(path):
    model = load_model()
    image = create_img(path)
    ans = model.predict(image)
    count = np.sum(ans)
    return count,image,ans

# Crowd Status
def frequency_status(count, limit):
    status = ['No Crowd', 'Sparse Crowd', 'Medium Dense Crowd', 'Dense Crowd']
    freq = ['Low', 'Medium', 'High']
    crowd_status = None
    crowd_freq = None

    # Crowd Status Metrics
    percentage1 = int(count * 0.2)
    percentage2 = int(count * 0.4)
    percentage3 = int(count * 0.6)
    percentage4 = int(count * 0.8)
    percentage5 = int(count * 1)
    
    # Distinct-dissimilar / No Crowd
    if count == 0 or limit <= percentage1:
        crowd_status = status[0]
        crowd_freq = freq[0]
        print(f'\nCrowd Status:\t {crowd_status}')
        print(f'\nCrowd Density Frequency:\t {crowd_freq}')
        
    # Low similarity / Sparse Crowd
    elif percentage1 < limit <= percentage2:
        crowd_status = status[1]
        crowd_freq = freq[0]
        print(f'\nCrowd Status:\t {crowd_status}')
        print(f'\nCrowd Density Frequency:\t {crowd_freq}')
        
    # Medium similarity / Medium Dense Crowd
    elif percentage2 < limit <= percentage3:
        crowd_status = status[2]
        crowd_freq = freq[1]
        print(f'\nCrowd Status:\t {crowd_status}')
        print(f'\nCrowd Density Frequency:\t {crowd_freq}')
        
    # High similarity / Dense Crowd
    elif percentage4 < limit <= percentage5 or limit > percentage5:
        crowd_status = status[3]
        crowd_freq = freq[2]
        print(f'\nCrowd Status:\t {crowd_status}')
        print(f'\nCrowd Density Frequency:\t {crowd_freq}')   
    else:
        print('\nPlease try different input values...')
